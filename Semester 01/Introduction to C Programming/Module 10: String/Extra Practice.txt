	string S, T;

}


A. Square or rectangle: https://codeforces.com/group/MWSDmqGsZm/contest/329103/problem/A
#include <bits/stdc++.h>
using namespace std;

int main() {
	int t, w, h;
	cin >> t;
	while(t--){
	    cin >> w >> h;
	    if(w==h){ cout << "Square" <<endl;}
	    else {cout << "Rectangle" <<endl;}
	}
    return 0;
}
