M. Replace MinMax: http://codeforces.com/group/MWSDmqGsZm/contest/219774/problem/M
#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;

    int A[N];
    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }

    int minPos = 0, maxPos = 0;

    for (int i = 1; i < N; i++) {
        if (A[i] < A[minPos]) minPos = i;
        if (A[i] > A[maxPos]) maxPos = i;
    }

    swap(A[minPos], A[maxPos]);

    for (int i = 0; i < N; i++) {
        cout << A[i] << " ";
    }

    return 0;
}

Alternative:

#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);

    int A[N];
    for (int i = 0; i < N; i++) {
        scanf("%d", &A[i]);
    }

    int minPos = 0, maxPos = 0;

    for (int i = 1; i < N; i++) {
        if (A[i] < A[minPos]) minPos = i;
        if (A[i] > A[maxPos]) maxPos = i;
    }

    int temp = A[minPos];
    A[minPos] = A[maxPos];
    A[maxPos] = temp;

    for (int i = 0; i < N; i++) {
        printf("%d ", A[i]);
    }

    return 0;
}
